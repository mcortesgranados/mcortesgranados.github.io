Implementing Collection and HashSet Using HashMap in Java

1. Implementing a Collection with HashMap in Java

Here's how to create a generic collection class backed by a HashMap:

        
import java.util.HashMap;
import java.util.Map;

public class MyCollection<T> {
    private Map<T, Boolean> map = new HashMap<>();

    // Add an element to the collection
    public void add(T element) {
        map.put(element, true);
    }

    // Check if the collection contains an element
    public boolean contains(T element) {
        return map.containsKey(element);
    }

    // Remove an element from the collection
    public void remove(T element) {
        map.remove(element);
    }

    // Get the size of the collection
    public int size() {
        return map.size();
    }

    // Clear the collection
    public void clear() {
        map.clear();
    }

    // Check if the collection is empty
    public boolean isEmpty() {
        return map.isEmpty();
    }
}

2. Implementing a HashSet Using a HashMap in Java

Since a HashSet is simply a collection of unique elements, we can implement it using a HashMap, where we only care about the keys and not the values. Hereâ€™s how to do it:

        
import java.util.HashMap;
import java.util.Map;

public class MyHashSet<T> {
    private Map<T, Boolean> map = new HashMap<>();

    // Add an element to the set
    public void add(T element) {
        map.put(element, true); // Value can be anything, only key is used
    }

    // Check if the set contains an element
    public boolean contains(T element) {
        return map.containsKey(element);
    }

    // Remove an element from the set
    public void remove(T element) {
        map.remove(element);
    }

    // Get the size of the set
    public int size() {
        return map.size();
    }

    // Clear the set
    public void clear() {
        map.clear();
    }

    // Check if the set is empty
    public boolean isEmpty() {
        return map.isEmpty();
    }
}
